<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="vertical" viewSourceURL="srcview/index.html">
<mx:Script>
    <![CDATA[
        
        private function handleKey(event:KeyboardEvent):void{
            output.text += " ("
            if(event.ctrlKey) output.text += "[c]";
            if(event.altKey) output.text += "[a]";
            output.text += event.keyCode + "," + event.keyLocation + ")";

        }
    ]]>
</mx:Script>
    <mx:TextArea editable="true" id="input" keyDown="handleKey(event)" width="640" height="480"/>
    <mx:TextArea id="output" editable="false"  width="640" height="480" />
</mx:Application>
